SRV1-COD:
Apt-get update && apt-get install freeradius freeradius-utils -y
systemctl enable –now radiusd
Nano /etc/raddb/clients.conf
“Все закомментировать и после строчки #format is still accepted”
Client ALL {
	  Ipaddr = 0.0.0.0/0
	 Secret = P@ssw0rd
} 
Ctrl+O next Ctrl+X


Nano /etc/raddb/users
“В самом конце файла” 
Netuser Cleartext-Password := “P@ssw0rd”
	    Service-Type = Administrative-User
              Cisco-AVPair = “shell:roles=admin”
Ctrl+O next Ctrl+X



Systemctl restart radiusd
“Открываем порт для работы”
nano /etc/raddb/sites-enabled/default
Ищем строку (она будет закомментирована, надо открыть и прописать)
interface = ens192.3




Добавим просто пользователя 
useradd netuser
Проверяем просто Радиус
radtest netuser P@ssw0rd 10.0.30.100 1812 P@ssw0rd
